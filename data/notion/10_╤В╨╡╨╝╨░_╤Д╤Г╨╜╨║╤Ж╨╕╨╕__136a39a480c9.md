# Выводит результат арифметической операции между полученными числами

def calculate(x, y, operator):
    if operator == "+":
        print(f"{x} + {y} = {x + y}")
    elif operator == "-":
        print(f"{x} - {y} = {x - y}")
    elif operator == "*":
        print(f"{x} * {y} = {x * y}")
    elif operator == "/":
        print(f"{x} / {y} = {x / y}")
    else:
        print("Проверьте введенный оператор!")


a = int(input("Введите первое целое число: "))
b = int(input("Введите второе целое число: "))
z = input("Введите арифметический оператор +, -, * или /: ")
calculate(a, b, z)
