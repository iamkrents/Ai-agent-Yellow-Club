# Срезы ====================================

name = "Yellow"

# Объяснение вместе
# Вносим в квадратные скобки 1 и последний символ
print(name[1:9])
# Отличие индексов от срезов
print(name[0])
print(name[0:10])
print(name[0:])
print(name[:9])
print(name[:])

# Задача самостоятельная
# В базе данных Yellowclub все курсы написаны в таком формате: "Yellowclub - название курса"
# Необходимо извлечь из названий курсов "Yellowclub - Python", "Yellowclub - Photoshop" только сами названия
course1 = "Yellowclub - Python"
course2 = "Yellowclub - Photoshop"
print(course1[13:])
print(course2[13:])
print("Yellowclub - Python"[13:])


# Шаги ==================================
# Добавление шагов в срез
# строки[старт:стоп:шаг]
print(name[0:10:2])
print(name[::2])
print(name[1:9:2])

# Отрицательный шаг
print(name[::-1])
