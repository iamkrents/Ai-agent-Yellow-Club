	**Ожидаемый результат**
	```python
# Какие есть операторы сравнения:
# > - больше
# < - меньше
# >= - больше или равно
# <= - меньше либо равно
# == - равно ("=" - это знак присвоения переменной)
# != - не равно

5 > 3

# Посмотреть на результат в консоли
print(5 > 3)
print(8 < 4)
print(8 >= 8)
print(1 <= 0)
print(5 == 5)
print(5 != 5)

# Проверить тип данных, рассказать про Boolean
type(True)
print(type(True))


# Операторы сравнения. работа с переменными
a = 5
print(a > 3)
b = 3
print(a == b)

# Можно расскахать, что str тоже можно сравнивать
a = "а"  # буквы оценивают по порядковому номеру: а - 1
b = "б"  # буквы оценивают по порядковому номеру: б - 2
print(a == b)
print(a != b)
print(a > b)
print(a < b)
	```
