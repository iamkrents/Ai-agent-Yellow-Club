// Срезы ====================================
// Срезы, попросту говоря это несколько символов из слова, допустим Привет, 
val name = "Константин"

// Объяснение вместе
// Вносим в квадратные скобки 1 и последний символ
print(name.substring(1, 9))
// Отличие индексов от срезов
println(name[0])
println(name.substring(0, 4)) // вводим начальный индекс и конечный
//результат Конст
println(name.substring(0..5)) // через перечисление
println(name.substring(5))

// Задача самостоятельная
// В базе данных Yellowclub все курсы написаны в таком формате: "Yellowclub - название курса"
// Необходимо извлечь из названий курсов "Yellowclub - Python", "Yellowclub - Photoshop" только сами названия
val course1 = "Yellowclub - Python"
val course2 = "Yellowclub - Photoshop"
println(course1.substring(13))
println(course2.substring(13))
println("Yellowclub - Python".substring(13))

// Методы строк ==========================

// Методы строк s.uppercase() и s.lowercase()
print(name.uppercase())
print(name.lowercase())

// Задача самостоятельная
// Получить имя пользователя, написанное с маленькой буквы (в нижнем регистре)
// Вывести на экран введенное имя заглавными буквами (в верхнем регистре)
println("Введите ваше имя c маленькой буквы: ")
val name = readln()
print(name.uppercase())
print(name.lowercase())

// Продолжение задачи вместе
// Вывести на экран введенное имя с первой заглавной буквой (первая буква в верхнем регистре)
val first_letter = name[0]
val first_letter = first_letter.uppercaseChar() // - это спец метод для букв, чтобы сделать их верхним регистром
# Упрощаем
first_letter = name[0].uppercaseChar()
print(first_letter)
val second_part = name.substring(1)
println(second_part)
println(first_letter + second_part)
// Решение в одну строку
println(name[0].uppercaseChar() + name.substring(1))
println(name.capitalize())

// Задача самостоятельная
// Получить 2 строки с разными словами

print("Введите 1-е слово: ")

print("Введите 1-е слово: ")
val st1 = readln()

val st2 = readln()
print(st1.lower() == st2.lower())
