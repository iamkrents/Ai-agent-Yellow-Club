	**Ожидаемый результат**
	// Вывод арифметических операций в консоль<br>console.log(3 + 5);<br>console.log(10 - 3);<br>console.log(2 \* 4);<br>console.log(15 / 3);
	// Проверка арифметических операций с переменными<br>let a = 10;<br>console.log(a \* 2);<br>let b = 5;<br>console.log(a / b);
	// Алгоритм ввода двух чисел и сложения<br>let aInput = prompt("Введите первое число: ");<br>let bInput = prompt("Введите второе число: ");<br>console.log("a + b =", aInput + bInput); // Это будет строковая конкатенация
	// Типы данных в JavaScript:<br>// number - целые и дробные числа (int и float объединены)<br>// string - текст в кавычках<br>// boolean - логический тип true / false
	// Фиксируем арифметические действия с number<br>a = 10;<br>b = 5;<br>console.log(a + b);<br>console.log(a - b);<br>console.log(a \* b);<br>console.log(a / b);
	// Конкатенация строк<br>let str1 = "Всем";<br>let str2 = "привет";<br>console.log(str1 + str2);
	// Умножение строки на число<br>let str = "Привет";<br>let times = 3;<br>console.log(str.repeat(times)); // аналог строки \* число в Python
	// Как узнать тип данных<br>let num = 2;<br>let floatNum = 3.2;<br>let text = "Yellow";<br>console.log(typeof num);<br>console.log(typeof floatNum);<br>console.log(typeof text);
	// Конвертация типов данных<br>// int(), float(), str() → в JS:<br>console.log(parseFloat("5")); // Пример преобразования строки в float<br>console.log(String(123));     // Преобразование числа в строку<br>console.log(Number("10.5"));  // Преобразование строки в число
	// Калькулятор, который складывает два введённых числа<br>let n1 = Number(prompt("Введите первое число: "));<br>let n2 = Number(prompt("Введите второе число: "));<br>console.log("a + b =", n1 + n2);
