# Методы списков

# Какие методы строк помните?
# s.upper()
# s.lower()
# s.capitalize()
# s.isdigit()
# s.strip()
products = "шоколад, картошка, молоко, чай, печенье, хлеб"
print(products)

# Метод split()
# список создается из строки автоматически, разделителем служит знак пробела
print(products.split())
# в скобках можем передавать разделитель (указанный разделитель удаляется)
print(products.split(","))
# в качестве разделителя можно передавать несколько символов идущих друг за другом, в том числе пробел
print(products.split(", "))

# Понимаем что split не меняет изначальную строку
print(products)


# Учимся перезаписывать переменную
name = "Константин"
print(name)
# Python интерпретируется линейно, поэтому данные из name будут перезаписаны, если мы внесем новые данные
name = "Аман"
print(name)

# Разбираем перезапись на нашем примере
# Мы можем создать новую переменную, тогда данные в переменной products останутся неизменными
products_lst = products.split(", ")
# Мы можем перезаписать переменную products, тогда данные в переменной будут изменены
products = products.split(", ")
# Теперь в переменной products не строка, а список
print(products)
