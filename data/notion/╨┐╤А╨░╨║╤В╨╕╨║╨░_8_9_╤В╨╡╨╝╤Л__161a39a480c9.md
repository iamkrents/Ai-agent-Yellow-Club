fun searchNum(list: List<Int>, num: Int){
    var isFound = false
    var indexOfNum = 0
    // По умолчанию мы считаем, что число мы еще не нашли и индекс равен первому числу
    for(i in list){
        if(i == num){
            isFound = true
            indexOfNum = list.indexOf(i)
            break
        }
    }
    if(isFound){
        println("Ура, число есть в списке, его индекс: $indexOfNum")
    }else{
        println("Похоже, этого числа нет в списке(")
    }
}

	больше ли число своего соседа. Если представить в голове, то у нас есть пупырчатая пленка, на ней написаны числа, допустим 2 и 1.

fun sortList(list: MutableList<Int>): MutableList<Int> {
    // Внешний цикл: количество проходов равно числу элементов - 1
    // При одном шаге внешнего цикла, у нас полностью проходит весь внутренний цикл. Не забываем!
    var isSorted = false // Флаг для проверки, отсортирован ли список
    while (!isSorted) {
        isSorted = true // Считаем, что список отсортирован, пока не найдем нарушение
        for (j in 0..list.size - 2) { // list.size - 2 потому что в конце нам нужно проверить предпоследний и последний элементы, если бы было просто -1, то выскочила бы
        // ошибка из-за того, что мы вышли за границы списка
            if (list[j] > list[j + 1]) {
                // Меняем соседние элементы местами, если порядок нарушен
                val temp = list[j]
                list[j] = list[j + 1]
                list[j + 1] = temp`````````````````````````````````````````````````,  
                isSorted = false // Если был обмен, список еще не отсортирован
            }
        }
    }
    // Возвращаем отсортированный список из функции
    return list
}

fun twister(){
    // Список частей тела и цветов
    val bodyParts = listOf("левая рука", "правая рука", "левая нога", "правая нога")
    val colors = listOf("красный", "синий", "зеленый", "желтый")

    // Запрашиваем количество игроков
    println("Введите количество игроков:")
    val playersCount = readln().toInt()

    if (playersCount <= 0) {
        println("Некорректное количество игроков. Игра завершена.")
        return
        /*
            Небольшая вводная. Мы можем использовать return в функциях без возвращения данных чтобы просто
            выйти из функции, то есть прекратить ее в любом месте. В данном случае если было введено 0 или
            меньше игроков, то мы выводим сообщение об ошибке и заканчиваем функцию.
        */
    }
    println("Начинаем игру в Твистер!")

    //создаем цикл, который будет длиться до тех пор, пока игрок не напишет "выход"
    while(true){
        println("Нажмите Enter, чтобы сделать ход, или введите \"выход\", чтобы завершить игру.")

        if (input.lowercase() == "выход") {
            println("Игра завершена. Спасибо за игру!")
            break

fun getStrongPassword(): String{
    // Строка, включающая в себя все буквы, спец символы и цифры
    val symbols = "qwertyuiopasdfghjklzxcvbnm1234567890!@#$%^&*()"
    var strongPassword = ""
    // Цикл для того, чтобы добавить 12 символов к нашему пока что пустому паролю
    for (i in 1..12){
        // Выбираем случайный символ
        val randomSymbol = symbols.get(Random.nextInt(0, symbols.length - 1))
        // Случайно выбираем, будет ли он в верхнем регистре
        val isUpper = Random.nextBoolean()
        // Проверяем, если isUpper true, то добавляем символ в верхнем регистре, если нет, то просто добавляем
        if(isUpper){
            strongPassword += randomSymbol.uppercase()
        }else{
            strongPassword += randomSymbol
        }
    }
    // Возвращаем наш сложный пароль
    return strongPassword
}
