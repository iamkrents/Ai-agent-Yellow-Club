/*
	Создайте программу, которая генерирует список из 5 случайных чисел от 0 до 9 и выводит его на экран. 
	Затем пользователь вводит число для поиска. Если число есть в списке, программа выводит его индекс, если нет — сообщает об этом. 
	Помните, что нумерация элементов начинается с 0.
*/

import random  # подключаем библиотеку для создания случайных чисел
numbers = []  # создаём пустой список

# добавляем в список 5 случайных чисел от 0 до 9
for i in range(5):
    numbers.append(random.randint(0, 9))
print("Наш список:", numbers)  # показываем получившийся список

# просим пользователя ввести число
num = int(input("Введите число, которое хотите найти: "))

# проверяем, есть ли это число в списке
if num in numbers:
    print("Ура! Число есть в списке.")  # если нашли
    print("Его индекс:", numbers.index(num))  # показываем позицию числа
else:
    print("Такого числа нет в списке.")  # если не нашли


/*
	Сортировка списка* (делаем вместе). Функция принимает изменяемый список чисел и возвращает его отсортированным по возрастанию.
	Сортировка будет производиться по методу пузырька. Суть в том, что мы будем переставлять числа с их соседями, исходя из того,
	больше ли число своего соседа. Если представить в голове, то у нас есть пупырчатая пленка, на ней написаны числа, допустим 2 и 1.
	Мы понимаем, что сначала должно идти 1, а потом 2. Поэтому мы лопаем пузырь 2 и на его место ставим пузырь 1. Соответственно на месте
	пузыря 1 будет пузырь 2, итог: пузырь 1 -> пузырь 2. Примерно то же смаое мы сейчас сделаем, но запрограммируя эти действия.
*/

import random

# функция сортировки пузырьком
def bubble_sort(numbers):
    n = len(numbers)  # узнаём длину списка
    # внешний цикл — сколько раз повторяем проход
    for i in range(n):
        # внутренний цикл — сравниваем соседние элементы
        for j in range(0, n - 1):
            if numbers[j] > numbers[j + 1]:
                # меняем элементы местами
                numbers[j], numbers[j + 1] = numbers[j + 1], numbers[j]
    return numbers
    
# создаём список из 5 случайных чисел
numbers = []
for i in range(5):
    numbers.append(random.randint(0, 9))
print("Список до сортировки:", numbers)

# сортируем
bubble_sort(numbers)
print("Список после сортировки:", numbers)


/*
	Твистер. Создаем функцию, которая запрашивает кол-во игроков и выводит условия для рандомного игрока, какую часть тела на какой цвет 
	поставить. Части тела и цвета лежат в отдельных списках. Игра идет до тех пор, пока один из игроков не напишет "выход".
*/

import random

def twister():
    # Списки частей тела и цветов
    body_parts = ["левая рука", "правая рука", "левая нога", "правая нога"]
    colors = ["красный", "синий", "зелёный", "жёлтый"]

    # Запрашиваем количество игроков
    players_count = int(input("Введите количество игроков: "))
    if players_count <= 0:
        print("Некорректное количество игроков. Игра завершена.")
        return

    print("Начинаем игру в Твистер!")

    # Игра продолжается до тех пор, пока кто-то не введёт "выход"
    while True:
        command = input('Нажмите Enter, чтобы сделать ход, или введите "выход" для завершения игры: ')
        if command.lower() == "выход":
            print("Игра завершена. Спасибо за игру!")
            break

        # Выбираем случайного игрока, часть тела и цвет
        random_player = "Игрок " + str(random.randint(1, players_count))
        random_body_part = random.choice(body_parts)
        random_color = random.choice(colors)

        print(f"{random_player} должен поставить {random_body_part} на {random_color}!")

# Запуск игры
twister()


/*
	Сложный пароль. Написать функцию, которая генерирует сложный пароль, чтобы нас не смогли взломать. 
	Сложным паролем будем считать строку длиной 12 символов, включающую в себя буквы разных регистров, спец символы (!"№;%:?*) и цифры. 
	Мы должны рандомно выбирать символ и его регистр и добавлять к изначально пустой строке.
*/

import random  # Подключаем библиотеку для работы со случайными числами и выборами

def get_strong_password():
    # Все символы, которые могут быть в пароле: буквы, цифры и спецсимволы
    symbols = "abcdefghijklmnopqrstuvwxyz1234567890!\"№;%:?*)"
    
    password = ""  # Сначала пароль пустой

    # Цикл для добавления 12 символов к паролю
    for _ in range(12):
        char = random.choice(symbols)  # Выбираем случайный символ из списка symbols

        # Случайно решаем, будет ли символ заглавной буквой
        if random.choice([True, False]):
            char = char.upper()  # Делаем символ заглавным

        password += char  # Добавляем символ к паролю

    return password  # Возвращаем готовый пароль

# Генерируем пароль и выводим его на экран
print(get_strong_password())
