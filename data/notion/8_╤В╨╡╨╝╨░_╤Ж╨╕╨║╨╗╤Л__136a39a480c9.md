score_lst = []
while len(score_lst) < 5:
    score = int(input("Введите оценку ученика: "))

# range() - числовой диапазон

# 1. range(10) - range(стоп)
r1 = range(10)
print(r1)
print(list(r1))

print(list(r2))

r3 = range(3, 10, 3)
print(r3)
print(list(r3))

# После каждого раунда можно забрать деньги
from random import randint

level = 1
money = 10
while money <= 1000000:
    print(f"Раунд - {level}. Вы можете выиграть {money} рублей!")
    random_number = randint(1, 2)
    your_number = int(input("Угадайте число от 1 до 2: "))
    if random_number == your_number:
        print(f"Поздравляем, вы выиграли! Ваш баланс {money} рублей!")
        if money == 1_000_000:
            print(f"Вы стали миллионером!".upper())
            break
        choice = input("Хотите продолжить или забрать деньги? (y / n): ")
        if choice == "n":
            print(f"Поздравляем, вы забираете {money} рублей!")
            break
        else:
            print("Отлично! Продолжаем! ")
            money *= 10
            level += 1
    else:
        print(f"Выпало число {random_number}!")
        print("К сожалению, вы проиграли... Ваш баланс 0 рублей...")
        choice = input("Хотите начать заново? (y / n): ")
        if choice == "n":
            print("Спасибо за игру! Удачи!")
            break
        else:
            print("Отлично! Продолжаем!")
            level = 1
            money = 10
