/*
Функция в Kotlin — это блок кода, который выполняет определённое действие и может быть вызван в программе, чтобы этот код сработал.
Функции позволяют группировать логику, переиспользовать её и писать код более компактно и понятно. То есть не нужно по несколько раз
писать одно и то же действие.
Функция объявляется ключевым словом fun. Синтаксис функции такой fun названиеФункции(){ тело функции }. Сама функция пишется вне блока
fun main(){}, а допустим снизу.

Пример: давайте напишем функцию, которая будет выводить приветствие:
*/

fun greeting(){
	println("Привет!")
}

/*
	Пробуем запустить код и видим, что ничего не происходит. Спрашиваем, может кто-то догадывается, что нужно сделать. Что же нужно 
	сделать: мы вызываем эту функцию в блоке (функции) main() -> fun main(){ greeting() }. Пробуем запустить снова и смотрим на результат.
	
	В функции так же можно создавать переменные и делать всё то, что делали до этого. А потом вызывать ее в главном блоке.
	Попробуем создать функцию, где пользователя просят ввести имя и год рождения, а после этого пытаются посчитать, сколько ему лет.
*/

fun calculateAge(){
	println("Введите имя: ")
	val name = readln()
	println("Введите год рождения: ")
	val year = readln().toInt()
	println("Вам ${2024 - year} лет?")
}

fun main(){ calculateAge() }

/*
	Функции с параметрами. В андроид приложении нет такого понятия, как ввод данных с консоли. Но в функцию же нужно как-то передавать
	ввод пользователя или другие параметры, как это можно сделать? Для этого функция может принимать параметры, выглядит это так:
	fun названиеФункции(названиеПараметра: Тип данных, второй параметр: Тип данных){ тело функции }
	Давайте сделаем функцию калькулятора, которая принимает два числа типа Double и знак, действие которого нужно сделать. 
	Результат выводить на экран:
*/

fun calculate(firstNum: Double, secondNum: Double, action: Char){
	when(action){
		'*' -> { println(firstNum * secondNum) }
		'/' -> { println(firstNum / secondNum) }
		'+' -> { println(firstNum + secondNum) }
		'-' -> { println(firstNum - secondNum) }
		else -> {
			println("Неверное действие")
		}
	}
}
/*
	Параметры в функцию мы можем вписывать как сразу, так и замещая их переменными
*/
fun main() {
    calculate(1.0, 2.0, '/')
    // другой способ
    val firstNum = 14.0
    val secondNum = 7.6
    val action = '-'
    calculate(firstNum, secondNum, action)
}

	делать вместе, сначала спрашивая у ребят, что нужно писать.
*/

fun main() {

/*
	Функции с возвращемым значением. Функция с возвращаемым результатом в программировании используется для выполнения 
	вычислений или операций, результат которых затем можно использовать в других частях программы.
	Синтаксис этой функции такой: 
	fun названиеФункции(названиеПараметра: Тип данных) : возвращаемый тип данных { 
	тело функции
	ключевое слово return данные нужного типа
	}
	Пример, давайте создадим функцию подсчета суммы списка, в функцию мы передаем список чисел, а возвращаем сумму
*/

fun sumOfList(list: List<Int>): Int {
    var sum = 0
    for(i in list){
        sum += i
    }
    return sum
}
/*
	Помимо возвращения результата через return, если функция маленькая, например сложить 2 числа, мы можем переписать ее так
	fun sumOfTwo(first: Int, second: Int) = first + second
	То есть мы заменяем тело функции и оператоор return на =
*/

fun main(){
	// Здесь либо передаем существующий список, либо вписываем сразу в функцию 
	val list = listOf<Int>(12,2,34,6)
  sumOfList(listOf(1,2,3,4))
  sumOfList(list)
}

/* 
	Просим запустить, понимаем, что ничего не выводится, задаем вопрос, почему. Объясняем, что возвращение результата это не вывод
	на экран. Это просто передача информации, а что с ней делать дальше, решает программист. Чтобы вывести результат на экран
	нужно просто поместить функцию в println(sumOfList(list))
*/

/*
	Также, результат можно поместить в переменную и дальше с ней что-то делать. Вот так:
*/

fun main() {
    val list = listOf<Int>(12,2,33,6)
    val result = sumOfList(list)
    if(result % 2 == 0){
        println("Результат четный")
    }else{
        println("Результат нечетный")
    }
}

/*
	Задача 1. Сумма чисел до введенного. Написать функцию, которая выводит на экран сумму всех чисел до полученного функцией.
	То есть функция получает 3, значит нужно вывести сумму 1+2. 
*/

fun sum(num: Int){
	var sum = 0
	for(num in 1..num - 1){ // -1 для того, чтобы не включать само переданное число
		sum += num
	}
	println("Сумма = $sum")
}

fun main(){
	sum(10)
}

/*
	Задача 2. Квадрат числа. Написать функцию, которая принимает целое число и возвращает его квадрат. Пользователь вводит число с
	клавиатуры, передает в функцию, а результат помещается в переменную. Результат вывести на экран 3 разa через пробел.
*/

fun square(num: Int) = num * num

fun main(){
	val num = readln().toInt()
	val res = square(num)
	for(i in 1..3){
		println(res)
	}

/*
	Задача 3*. Максимальный элемент списка. Найти в списке максимальный элемент, вывести его на экран. В функцию приходит список чисел,
	она возвращает одно наибольшее число.
*/

fun findMax(list: List<Int>) {
	var maxNum = 0
	for(num in list){
		if(num > maxNum){
			maxNum = num
		}
	}
	println(maxNum)
}

fun main(){
    findMax(listOf(1,2,7,4,56,99,23,11,0))
}

/*
	Логика: максимальное число у нас не известно, поэтому оно будет 0, а дальше в цикле каждое число списка сравнивается с 
	"максимальным", если число из списка больше, то теперь оно - максимальное, и эта процедура будет повторять каждый шаг цикла.
	Если же число списка меньше, то цикл просто идет дальше
*/
