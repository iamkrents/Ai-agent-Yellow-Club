# 1. Добавляем переменные для игры
Сразу после кода, который рисует голову змейки, добавим две новые переменные.
```c++
bool game_running = true;
int sleep_time = 120;
```
### Что они делают
**game_running**
Это специальная переменная-флаг.
Она отвечает за вопрос:
> Игра ещё продолжается или уже закончилась?
Если значение `true` — игра продолжается.
Если `false` — игра завершается.
---
**sleep_time**
Эта переменная задаёт **скорость змейки**.
Функция `Sleep()` работает в миллисекундах.
```plain text
1000 миллисекунд = 1 секунда
```
Если поставить:
```plain text
Sleep(120)
```
программа будет делать **паузу 0.12 секунды между кадрами**.
Если уменьшить число — змейка станет быстрее.
---
# 2. Создаём главный цикл игры
Теперь добавим главный цикл:
```c++
while (game_running) {
```
Этот цикл означает:
> Выполняй код снова и снова, пока игра не закончится.
Это **сердце любой игры**.
Каждый проход цикла — это **один шаг игры**.
Внутри цикла происходит:
- проверка клавиш
- движение змейки
- проверка столкновений
- обновление картинки
---
# 3. Добавляем паузу между шагами
Внутри цикла сразу пишем:
```c++
Sleep(sleep_time);
```
Это нужно для того, чтобы:
- программа **не выполнялась слишком быстро**
- змейка **двигалась плавно**
Без этой строки змейка перемещалась бы **тысячи раз в секунду**.
---
# 4. Проверяем нажатие клавиш
Теперь нужно научить программу **реагировать на стрелки клавиатуры**.
Для этого используется функция:
```plain text
_kbhit()
```
Она проверяет:
> Нажал ли пользователь клавишу.
Если нажал — мы можем считать её код.
```c++
if (_kbhit()) {
    int k = _getch();
```
`_getch()` считывает код нажатой клавиши.
---
# 5. Почему стрелки читаются два раза
У стрелок клавиатуры есть особенность.
Когда нажимается стрелка, компьютер отправляет **два кода**:
```plain text
0 или 224
потом настоящий код стрелки
```
Поэтому мы проверяем:
```c++
if (k == 0 || k == 224) {
    k = _getch();
}
```
Теперь переменная `k` содержит **реальный код клавиши**.
---
# 6. Определяем направление движения
Для обработки клавиш используем `switch`.
```c++
switch (k)
```
Здесь мы проверяем **какая именно клавиша была нажата**.
### Стрелка вверх
```c++
case 72:
```
Меняем направление движения:
```c++
dy = -1;
dx = 0;
```
Это значит:
- по X не двигаемся
- по Y идём вверх
---
### Стрелка вниз
```c++
case 80:
dy = 1;
dx = 0;
```
---
### Стрелка влево
```c++
case 75:
dx = -1;
dy = 0;
```
---
### Стрелка вправо
```c++
case 77:
dx = 1;
dy = 0;
```
---
### Клавиша ESC
```c++
case 27:
game_running = false;
```
Это завершает игру.
---
# 7. Считаем новую позицию головы
Теперь нужно понять:
> куда змейка переместится на следующем шаге.
```c++
int new_x = array_X[0] + dx;
int new_y = array_Y[0] + dy;
```
Если направление вправо (`dx = 1`), координата X увеличится на 1.
---
# 8. Проверяем столкновение со стеной
Если координаты головы выходят за границу поля — игра заканчивается.
```c++
if (new_x <= 0 || new_x >= width - 1 ||
    new_y <= 0 || new_y >= height - 1)
{
    game_running = false;
}
```
Это означает:
- если змейка коснулась рамки
- игра заканчивается
---
# 9. Перемещаем голову змейки
Сначала нужно **стереть старую голову**.
```c++
c.X = array_X[0];
c.Y = array_Y[0];
SetConsoleCursorPosition(h, c);
cout << snake;
```
Голова превращается в **обычную часть тела**.
---
Теперь обновляем координаты головы:
```c++
array_X[0] = new_x;
array_Y[0] = new_y;
```
---
И рисуем голову на новом месте:
```c++
c.X = new_x;
c.Y = new_y;
SetConsoleCursorPosition(h, c);
cout << head;
```
---
# Итог
Сегодня мы добавили **главную механику любой игры**:
- игровой цикл
- управление клавиатурой
- движение объекта
- проверку столкновений
Теперь у нас есть **настоящая основа игры Snake**.

# Полный код после занятия
```c++
#include <iostream>
#include <windows.h>
#include <conio.h>
using namespace std;

int main() {
    srand(time(0));
    system("title Snake Game");
    system("mode con cols=70 lines=31");
    MoveWindow(GetConsoleWindow(), 50, 50, 800, 600, true);

    const int width = 50, height = 30;
    const int max_length = 50;

    int array_X[max_length];
    int array_Y[max_length];
    int length = 1;

    array_X[0] = width / 2;
    array_Y[0] = height / 2;

    int dx = 1, dy = 0;         // ← начинаем двигаться вправо

    int X_apple, Y_apple;
    do {
        X_apple = rand() % (width - 2) + 1;
        Y_apple = rand() % (height - 2) + 1;
    } while (X_apple == array_X[0] && Y_apple == array_Y[0]);

    char snake = '*';
    char head = 2;
    char apple = 'o';

    COORD c;
    HANDLE h = GetStdHandle(STD_OUTPUT_HANDLE);

    CONSOLE_CURSOR_INFO cci = { sizeof(cci), false };
    SetConsoleCursorInfo(h, &cci);

    // ─────────────── Рамка и начальная картинка (это уже было на прошлом уроке) ───────────────
    SetConsoleTextAttribute(h, 4);
    for (int y = 0; y < height; y++) {
        for (int x = 0; x < width; x++) {
            if (x == 0 || x == width - 1 || y == 0 || y == height - 1)
                cout << '#';
            else
                cout << ' ';
        }
        cout << "\n";
    }

    c.X = X_apple; c.Y = Y_apple;
    SetConsoleCursorPosition(h, c);
    SetConsoleTextAttribute(h, 12);
    cout << apple;

    c.X = array_X[0]; c.Y = array_Y[0];
    SetConsoleCursorPosition(h, c);
    SetConsoleTextAttribute(h, 10);
    cout << head;

    // ───────────────────────────────────────────────────────────────
    // ↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓  НОВЫЙ КОД — НАЧИНАЕМ ПИСАТЬ ЗДЕСЬ  ↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓
    // ───────────────────────────────────────────────────────────────

    bool game_running = true;           // флаг "игра ещё идёт"
    int sleep_time = 120;               // сколько миллисекунд ждать между шагами (чем меньше — тем быстрее)

    // Самый главный цикл игры — будет крутиться, пока не проиграем
    while (game_running) {

        Sleep(sleep_time);              // ← делаем маленькую паузу, чтобы змейка не летала как ракета

        // ─── Проверяем, нажал ли игрок клавишу ───────────────────────────────
        if (_kbhit()) {                 // если в очереди есть хоть одна клавиша
            int k = _getch();           // берём первую часть кода

            // стрелки дают два байта — иногда сначала 0 или 224, потом настоящий код
            if (k == 0 || k == 224) {
                k = _getch();           // берём вторую часть
            }

            // теперь смотрим, какая именно клавиша
            switch (k) {
            case 72:    // стрелка вверх
                if (dy != 1) {      // запрещаем разворот на 180° (нельзя сразу вниз)
                    dy = -1;
                    dx = 0;
                }
                break;

            case 80:    // вниз
                if (dy != -1) {
                    dy = 1;
                    dx = 0;
                }
                break;

            case 75:    // влево
                if (dx != 1) {
                    dx = -1;
                    dy = 0;
                }
                break;

            case 77:    // вправо
                if (dx != -1) {
                    dx = 1;
                    dy = 0;
                }
                break;

            case 27:    // ESC — выход
                game_running = false;
                break;
            }
        }

        // ─── Считаем, куда сейчас пойдёт голова ───────────────────────────────
        int new_x = array_X[0] + dx;
        int new_y = array_Y[0] + dy;

        // ─── Проверяем, не врезались ли мы в стену ───────────────────────────────
        if (new_x <= 0 || new_x >= width - 1 ||
            new_y <= 0 || new_y >= height - 1) {
            game_running = false;       // конец игры
            continue;                   // сразу выходим из этой итерации цикла
        }

        // ─── Двигаем голову (пока без тела) ───────────────────────────────
        // затираем старую голову — делаем её обычным кусочком тела
        c.X = array_X[0];
        c.Y = array_Y[0];
        SetConsoleCursorPosition(h, c);
        cout << snake;                  // была голова — стала тело

        // обновляем координаты головы
        array_X[0] = new_x;
        array_Y[0] = new_y;

        // рисуем новую голову
        c.X = new_x;
        c.Y = new_y;
        SetConsoleCursorPosition(h, c);
        cout << head;
    }

    // ───────────────────────────────────────────────────────────────
    // ↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑  КОНЕЦ НОВОГО КОДА  ↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑
    // ───────────────────────────────────────────────────────────────

    system("cls");
    SetConsoleTextAttribute(h, 7);
    cout << "GAME OVER\n\n";
    cout << "Нажми любую клавишу, чтобы выйти...\n";

    return 0;
}
```
