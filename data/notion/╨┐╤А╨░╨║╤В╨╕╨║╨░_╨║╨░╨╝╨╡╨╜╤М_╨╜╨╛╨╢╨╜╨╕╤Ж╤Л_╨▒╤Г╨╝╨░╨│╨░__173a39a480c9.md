import random
from tkinter import *

# 4 Интерфейс ==============================
# 5 Расказать про ООП
root = Tk()

# 7 Указать титул
root.title('Камень ножницы бумага')
# 8 Указать геометрию
root.geometry('600x400')
# 9 Заблокировать геометрию
root.resizable(width=False, height=False)
# 10 Указать цвет бэкраунда
root['bg'] = '#000'

# 11 Добавить текстовые поля
user_text = Label(root, text='Игрок: 0', font=20, bg='white')
user_text.place(x=50, y=100)

computer_text = Label(root, text='Компьютер: 0', font=20, bg='white')
computer_text.place(x=50, y=150)

win_text = Label(root, text='Выиграл - ', font=20, bg='white')
win_text.place(x=50, y=200)

computer_choice = Label(root, text='Компьютер выбрал - ', font=20, bg='white')
computer_choice.place(x=50, y=250)

# 13.3
user_count = 0
comp_count = 0


# 13.1 Создать функцию
def action_press(btn):
    lst = ['камень', 'ножницы', 'бумага']
    computer = random.choice(lst)
    user = btn['text']
    print(computer, user)
    global user_count
    global comp_count
		
		# 13.4 Написать алгоритм
    if user == computer:
        win_text.configure(text="Выиграл - Ничья")
    elif user == 'камень':
        if computer == 'ножницы':
            win_text.configure(text="Выиграл - Игрок")
            user_count += 1
            user_text.configure(text=f'Игрок: {user_count}')
        else:
            win_text.configure(text="Выиграл - Компьютер")
            comp_count += 1
            computer_text.configure(text=f'Компьютер: {comp_count}')
    elif user == 'ножницы':
        if computer == 'бумага':
            win_text.configure(text="Выйграл - Игрок")
            user_count += 1
            user_text.configure(text=f'Игрок: {user_count}')
        else:
            win_text.configure(text="Выйграл - Компьютер")
            comp_count += 1
            computer_text.configure(text=f'Компьютер: {comp_count}')
    elif user == 'бумага':
        if computer == 'камень':
            win_text.configure(text="Выйграл - Игрок")
            user_count += 1
            user_text.configure(text=f'Игрок: {user_count}')
        else:
            win_text.configure(text="Выйграл - Компьютер")
            comp_count += 1
            computer_text.configure(text=f'Компьютер: {comp_count}')
            
		# 13.5 Изменить текста выбора компьютера
    computer_choice.configure(text=f"Компьютер выбрал - {computer}")


# 12 Кнопки =======================================
# 12.1 Создать кнопки
# 12.2 Разместить кнопки
# 13.2 Функциянирование кнопок пишеться потом

stone = Button(root, text='камень', font=20, bg='white')
stone.config(command=lambda: action_press(stone))
stone.place(x=50, y=300)

scissor = Button(root, text='ножницы', font=20, bg='white')
scissor.config(command=lambda: action_press(scissor))
scissor.place(x=220, y=300)

paper = Button(root, text='бумага', font=20, bg='white')
paper.config(command=lambda: action_press(paper))
paper.place(x=420, y=300)

# 6 Запустить интерфейс
root.mainloop()
