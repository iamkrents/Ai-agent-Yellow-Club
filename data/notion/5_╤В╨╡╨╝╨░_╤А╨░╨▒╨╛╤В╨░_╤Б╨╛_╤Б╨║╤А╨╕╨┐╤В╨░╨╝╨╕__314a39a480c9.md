1. Добавляем wait()
```lua
-- Берём объект, внутри которого находится этот скрипт
local part = script.Parent  

-- Меняем цвет объекта
part.Color = Color3.fromRGB(255, 0, 0)

-- Ждём 5 секунд
wait(5)

```
1. Добавляем цикл
```lua
-- Берём объект, внутри которого находится этот скрипт
local part = script.Parent  

-- while true do означает:
-- "Повторяй всё, что внутри, снова и снова без остановки"
while true do
	
	-- Меняем цвет объекта
	part.Color = Color3.fromRGB(255, 0, 0)
	
	-- Ждём половину секунды (0.5 секунды)
	wait(0.5)
	
	-- Меняем цвет объекта
	part.Color = Color3.fromRGB(255, 46, 147)
	
	-- Снова ждём половину секунды
	wait(0.5)

-- После этого компьютер возвращается наверх и делает всё заново
end
```
1. Делаем смену дня и ночи
```lua
-- Обращаемся к службе Lighting (она отвечает за освещение и время суток в игре)
local sky = game:GetService("Lighting")

-- while true do означает:
-- "Повторяй всё, что внутри, снова и снова без остановки"
while true do
	
	-- Устанавливаем время суток
	sky.ClockTime = 14
	
	-- Ждём 5 секунд
	wait(5)
	
	-- Устанавливаем время суток
	sky.ClockTime = 0
	
	-- Снова ждём 5 секунд
	wait(5)

-- После этого компьютер возвращается наверх и делает всё заново
end
```
1. Плавная смена дня и ночи 
```lua
-- Обращаемся к службе Lighting (она отвечает за освещение и время суток в игре)
local lighting = game:GetService("Lighting")

-- Бесконечный цикл
-- Всё, что внутри, будет повторяться снова и снова
while true do
	
	-- Увеличиваем игровое время на маленькое число
	-- 0.01 — это очень небольшой шаг, поэтому время меняется плавно
	lighting.ClockTime += 0.01
	
	-- Проверяем: если время стало 24 или больше
	-- (24 — это конец суток в Roblox)
	if lighting.ClockTime >= 24 then
		
		-- Сбрасываем время на 0 (полночь)
		-- Так начинается новый игровой день
		lighting.ClockTime = 0
		
	end
	
	-- Делаем маленькую паузу,
	-- чтобы изменение времени было плавным и не слишком быстрым
	wait(0.1)

-- После этого цикл начинается заново
end
```

Итоговый правильный зполненный вариант
```lua
-- берём модель двери
local doorModel = script.Parent

-- находим ProximityPrompt внутри DoorPart
local prompt = doorModel.DoorPart.ProximityPrompt  -- DoorPart — часть, в которой лежит ProximityPrompt

-- название инструмента (ключа)
local requiredToolName = "Key"

-- состояние двери (закрыта = false)
local isOpen = false

-- защита от спама (чтобы нельзя было нажимать 100 раз подряд)
local debounce = false

-- угол открытия
local openAngle = 90   -- дверь будет открываться на 90 градусов

-- запоминаем закрытую позицию двери
local closedPivot = doorModel:GetPivot()

-- создаём позицию открытой двери (поворот)
local openPivot = closedPivot * CFrame.Angles(
	0,
	math.rad(openAngle),  -- переводим градусы в радианы
	0
)

---------------------------------------------------
-- ФУНКЦИЯ АНИМАЦИИ (плавное движение двери)
---------------------------------------------------

local function animateDoor(targetPivot)

	-- запоминаем откуда начинаем движение
	local startPivot = doorModel:GetPivot()

	-- цикл для плавного движения (30 шагов)
	for i = 1, 30 do
		
		local alpha = i / 30   -- число от 0 до 1 (для плавного перехода)
		
		-- плавно двигаем дверь от начальной позиции к целевой
		doorModel:PivotTo(startPivot:Lerp(targetPivot, alpha))
		
		task.wait(0.01) -- маленькая пауза для плавности
	end
end

---------------------------------------------------
-- СОБЫТИЕ НАЖАТИЯ
---------------------------------------------------

prompt.Triggered:Connect(function(player)

	-- если уже нажимали — выходим
	if debounce then return end
	debounce = true

	-- получаем персонажа игрока
	local character = player.Character

	-- ищем инструмент в руках (Tool внутри персонажа)
	local tool = character and character:FindFirstChildOfClass("Tool")

	---------------------------------------------------
	-- ПРОВЕРКА КЛЮЧА
	---------------------------------------------------

	-- если дверь закрыта И ключа нет → не открываем
	if not isOpen and (not tool or tool.Name ~= requiredToolName) then
		
		print("Нужен ключ!") -- выводим сообщение в Output
		
		debounce = false
		return
	end

	---------------------------------------------------
	-- МЕНЯЕМ СОСТОЯНИЕ
	---------------------------------------------------

	-- меняем состояние двери на противоположное
	isOpen = not isOpen

	-- выбираем куда двигать дверь
	local target = isOpen and openPivot or closedPivot

	-- запускаем анимацию
	animateDoor(target)

	task.wait(0.2)

	debounce = false
end)
```
