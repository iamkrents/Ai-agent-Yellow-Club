<callout icon="✅" color="gray_bg">
	**Ожидаемый результат урока**
	```kotlin
// Правила наименования переменных
// - можно использовать только латинские буквы
// - нельзя использовать цифры в начале
// - нельзя использовать символы кроме нижнего подчеркивания
// - использовать названия по смыслу

// Точка входа
fun main(){
	
	// Функция вывода данных
	print("Привет!")

	// Функция вывода данных и перевод на новую строку
	println("Как дела?")

	// Создаем переменную name и вносим туда свое имя, выводим в консоль
	var name = "Константин"
	println(name)

	// Разбираем разницу между val и var
	// var - переменные
	// val - константы

	// Самостоятельно создаем переменную возраста и выводим данные в консоль
	var age = 33
	println(age)

	// Перезаписываем переменную возраста и выводим данные в консоль
	age = 34
	println(age)

	// Самостоятельно работа
	// Выведите в консоль тект ниже (вместо троеточий должы подставляться данные из переменных):
	// Привет!
	// Меня зовут ... .
	// Мне ... лет.
	// В следующейм году будет ... .

	println("Привет!")
	val name = "Константин"
	var age = 33
	println("Меня зовут $name.")
	println("Мне $age года.")
	age = 34
	println("В следующем году будет $age.")
}
	```
</callout>
