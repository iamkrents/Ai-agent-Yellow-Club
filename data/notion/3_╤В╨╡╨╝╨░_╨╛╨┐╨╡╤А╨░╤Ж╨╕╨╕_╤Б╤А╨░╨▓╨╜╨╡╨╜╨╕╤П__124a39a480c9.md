	**Ожидаемый результат урока**
	```kotlin
// Какие есть операторы сравнения:
// > - больше
// < - меньше
// >= - больше или равно
// <= - меньше либо равно
// == - равно ("=" - это знак присвоения переменной)
// != - не равно

5 > 3

// Посмотреть на результат в консоли
println(5 > 3)
println(8 < 4)
println(8 >= 8)
println(1 <= 0)
println(5 == 5)
println(5 != 5)

// Тип данных Boolean - булевый тип данных true и false

// Операторы сравнения работа с переменными
a = 5
print(a > 3)
b = 3
print(a == b)

// Определение переменных с типом данных Boolean и запись значений в них
// Определение логической переменной должно быть таким
// - начинается на is
// - потом следует слово
var isMaxNum: Boolean = true

// Запись булевых значений в переменную
a = 5
b = 3
var isMaxNum: Boolean = a > b
println(isMaxNum)

// Знак ! указывает на то, что переменная будет с противоположным значением
println(!isMaxNum)

// Запросить ввести 1-е число через консоль
// Запросить ввести 2-е число через консоль
// Вывести в консоль результаты операций сравнения
println("Введите первое число:")
val num1 = readln().toInt()
println("Введите второе число:")
val num2 = readln().toInt()

println("Первое число больше второго: ${num1 > num2}")
println("Первое число меньше второго: ${num1 < num2}")
println("Равны ли ваши числа: ${num1 == num2}")

// Напишите алгоритм, который:
// запросит ввести логин
// запросит ввести пароль
// запросит повторно ввести пароль
// выведет в консоль фразу "Пароли совпадают: "
// выведет булевое значение как результат
// в алгоритме обязательно должна быть булевая переменная

print("Введите логин: ")
var login = readln()
print("Введите пароль: ")
var password1 = readln()
print("Повторите пароль: ")
var password2 = readln()
var isPasswordCorrect: Boolean = password1 == password2
println("Пароли совпадают: $isPasswordCorrect")
	```
