	**Ожидаемый результат**
	```python
/*
	Небольшое вступление, мы забыли про операцию деления с остатком, она не сложная, поэтому сразу закрепим на практике.
	Знак деления с остатком - %. Пример 3 % 2 = 1, то есть остаток.
*/
print(3 % 2)
print(4 % 2)
/*
	Четное или нечетное. Ввести с клавиатуры одно целое число. Проверить является ли число четным или нечетным, результат вывести на экран.
	Решение: Мы должны понять, что четным число считается, когда отстаток от деления на 2 равен 0, если остаток отличается, то число
	считается нечетным.
*/
num = int (input("Введите число: "))
if(num % 2 == 0):
	print("Четное")
else:
	print("Нечетное")

/*
	Задача 1. Арифметические операции. Палиндром. С клавиатуры вводится число до 99. Нужно проверить, является ли оно палиндромом. Палиндром - 
	это слово, фраза, число, которые читаются слева направо и справа налево одинаково. Пример: привет - не палиндром, шалаш - палиндром,
	99 - палиндром
*/

num = int(input("Введите двузначное число: ")
// Если остаток от деления равено результату деления на 10, то число палиндром 88 // 8 и 88 % 10 = 8
if num // 10 == num % 10:
	print(f"{num} - палиндром")
else:
	print(f"{num} - не палиндром")

/*
	Задача 2. Операции сравнения и логические операции. Программа просит ввести твой рост, потом рост твоего соседа. В результате
	на экране должен появиться ответ, выше ли ты своего соседа. (true, false)
*/

myHeight = int(input("Введи свой рост: "))
neighborHeight = int(input("Теперь введи рост твоего соседа: "))
print(f"Я выше? - {myHeight > neighborHeight}")

/*
	Задача 3. Условная конструкция if-elif-else. Программа просит ввести одного игрока камень, ножницы или бумагу, после просит то же самое
	второго игрока. Исходя из того, что ответили игроки, на экране должно появиться сообщение о том, что игрок1 или игрок2 победил.
	При желании, можно добавить возможность игрокам ввести их имена. Возможен вариант с многочисленными else-if, это не ошибка.
*/

player1_name = input("Игрок 1, введите свое имя: ")
player2_name = input("Игрок 2, введите свое имя: ")

player1_choice = input(f"{player1_name}, выберите камень, ножницы или бумагу: ").lower()
player2_choice = input(f"{player2_name}, выберите камень, ножницы или бумагу: ").lower()

if (player1_choice == "камень" and player2_choice == "ножницы") or \
   (player1_choice == "ножницы" and player2_choice == "бумага") or \
   (player1_choice == "бумага" and player2_choice == "камень"):
    print(f"{player1_name} победил!")
elif (player2_choice == "камень" and player1_choice == "ножницы") or \
     (player2_choice == "ножницы" and player1_choice == "бумага") or \
     (player2_choice == "бумага" and player1_choice == "камень"):
    print(f"{player2_name} победил!")
elif player1_choice == player2_choice:
    print("Ничья!")
else:
    print("Ошибка: неверный ввод. Начните заново.")

/*
	Задача 4. Списки. Сумма элементов списка. Создан массив с n-ым количеством целочисленных элементов. Задача сложить каждый элемент и вывести результат 
	на экран. Так как циклы мы еще не знаем, сумма будет вычисляться путем индексов массива и переменных.
*/
	// самое быстрое решение
arr = [1, 2, 4, 5]
total = arr[0] + arr[1] + arr[2] + arr[3]
print(total)
/*
	Возможный вариант решения:
	arr = [1,2,4,5]
	el1 = arr[0]
	el2 = arr[1]
	el3 = arr[2]
	el4 = arr[3]
	res = el1 + el2 + el3 + el4 
	print(res)
*/

/*
	Задача 5. Списки и методы списков. Список дел. Введи через запятую с пробелом список дел на сегодня. После чего,
	удали дело, которое ты уже сделал по индексу и еще одно по названию, а потом добавь одно, которое хотел бы успеть сделать. На экран вывести начальный список, список после 
	удаления сделанных дел, список после добавления дел.
*/

tasks = input("Введите список дел через запятую с пробелом: ").split(", ")
print("Начальный список:", tasks)

action_to_remove = input("Введите название дела, которое вы уже выполнили: ")
tasks.remove(action_to_remove)
print("После удаления по названию:", tasks)

action_to_remove_index = int(input("Введите индекс дела, которое вы уже выполнили: "))
tasks.pop(action_to_remove_index)
print("После удаления по индексу:", tasks)

action_to_add = input("Введите название нового дела: ")
tasks.append(action_to_add)
print("После добавления:", tasks)
	
/*
	Задача 6. Создание строки из последнего символа элементов массива. С клавиатуры через пробел вводится 4 любых слова,
	у каждого слова взять первую букву и соеденить их. Сделать проверку на размер списка. (Размер списка должен быть 4,
	если же он будет меньше 4, то появится ошибка)
	*/
words = input("Введите 4 слова через пробел: ").split(" ")

if len(words) == 4:  
    result = words[0][0] + words[1][0] + words[2][0] + words[3][0]
    print("Результат:", result)
else:
    print("Ошибка: нужно ввести ровно 4 слова.")
	```
