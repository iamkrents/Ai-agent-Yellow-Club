	```python
# Правила наименования переменных:
# - можно использовать только латинские буквы
# - нельзя использовать цифры в начале
# - нельзя использовать символы кроме нижнего подчеркивания

# Создаем переменную name и вносим туда свое имя, выводим в консоль
name = "Константин"
print(name)

# Самостоятельно создаем переменную возраста и выводим данные в консоль
age = 32
print(age)

# Самостоятельно делаем вывод множества значений в консоль
print("Привет")
print("Меня зовут", name)
print("Мне", age, "лет")
	```
	```python
# функция ввода данных
 input("Введите данные: ")

# сохранение введённых данных в переменную и вывод в консоль
 name = input("Введите имя: ")
 age = input("Введите возраст: ")
 print("Ваше имя: ", name)
 print("Ваш возраст: ", age)

# Сделайте программу, которая:
# запросит ввести логин и пароль
# выведет введенные данные на экран для проверки.
login = input("Введите логин: ")
password = input("Введите пароль: ")
print("Проверьте введенные данные: ")
print("Логин: ", login)
print("Пароль: ", password)
	```
