	```c++
#include <SFML/Graphics.hpp>  // Всё для окна, цветов, фигур, таймера и рисования
#include <ctime>              // time(0) для srand()
#include <optional>           // optional для pollEvent() в SFML3

using namespace sf;
using namespace std;

int main()
{
    // Инициализируем генератор случайных чисел,
    // чтобы яблоко появлялось каждый запуск в новом месте.
    srand(time(0));

    // Размеры игрового поля в КЛЕТКАХ, а не в пикселях.
    // width = ширина поля, height = высота поля.
    const int width = 50, height = 30;

    // Максимальная длина змейки.
    // Больше этого значения массивы уже не смогут хранить.
    const int max_length = 50;

    // Размер одной клетки в пикселях.
    // Например, если cell = 20, то одна игровая клетка
    // будет рисоваться как квадрат 20x20 пикселей.
    const float cell = 20.f;

    // Создаем графическое окно.
    // Размер окна считаем так:
    // ширина окна = width * cell
    // высота окна = height * cell
    RenderWindow window(
        VideoMode({ (unsigned int)(width * cell), (unsigned int)(height * cell) }),
        "Snake Game"
    );

    // Эти два массива хранят координаты всех частей змейки.
    // array_X[i] — X-координата i-го звена
    // array_Y[i] — Y-координата i-го звена
    int array_X[max_length];
    int array_Y[max_length];

    // Текущая длина змейки.
    // В начале у змейки только голова, поэтому длина = 1.
    int length = 1;

    // Начальная позиция головы — примерно центр поля.
    array_X[0] = width / 2;
    array_Y[0] = height / 2;

    // Направление движения змейки.
    // dx = изменение по X
    // dy = изменение по Y
    //
    // Сейчас:
    // dx = 1  -> двигаемся вправо
    // dy = 0  -> по Y не двигаемся
    int dx = 1, dy = 0;

    // Координаты яблока.
    int X_apple, Y_apple;

    // Ставим яблоко в случайное место внутри поля,
    // но не на стену и не на голову змейки.
    do {
        X_apple = rand() % (width - 2) + 1;
        Y_apple = rand() % (height - 2) + 1;
    } while (X_apple == array_X[0] && Y_apple == array_Y[0]);

    // Флаг работы игры.
    // true  -> игра идет
    // false -> игра окончена
    bool game_running = true;

    // timer накапливает прошедшее время.
    float timer = 0.f;

    // delay — задержка между шагами змейки.
    // 0.12 секунды = примерно 120 миллисекунд.
    // Чем меньше delay, тем быстрее движется змейка.
    float delay = 0.12f;

    // Таймер SFML для измерения прошедшего времени между кадрами.
    Clock clock;

    // Один прямоугольник, которым будем рисовать вообще всё:
    // рамку, яблоко, тело и голову.
    //
    // Почему cell - 1.f?
    // Чтобы между клетками был небольшой зазор, и поле выглядело аккуратнее.
    RectangleShape block(Vector2f(cell - 1.f, cell - 1.f));

    // Главный цикл программы:
    // работает, пока окно открыто.
    while (window.isOpen())
    {
        // =========================
        // 1. ОБРАБОТКА СОБЫТИЙ ОКНА
        // =========================

        // В SFML3 pollEvent() возвращает optional.
        // Пока события есть — обрабатываем их.
        while (const optional event = window.pollEvent())
        {
            // Если пользователь нажал на крестик окна — закрываем окно.
            if (event->is<Event::Closed>())
                window.close();
        }

        // Если игра закончилась, закрываем окно.
        // Позже можно будет заменить это на экран Game Over.
        if (!game_running)
        {
            window.close();
        }

        // =========================
        // 2. ВВОД С КЛАВИАТУРЫ
        // =========================

        // Движение вверх по клавише W.
        // Проверка dy != 1 не даёт мгновенно развернуться вниз,
        // иначе змейка могла бы "врезаться в себя" за один шаг.
        if (Keyboard::isKeyPressed(Keyboard::Key::W))
        {
            if (dy != 1) {
                dy = -1;
                dx = 0;
            }
        }

        // Движение вниз по клавише S.
        if (Keyboard::isKeyPressed(Keyboard::Key::S))
        {
            if (dy != -1) {
                dy = 1;
                dx = 0;
            }
        }

        // Движение влево по клавише A.
        if (Keyboard::isKeyPressed(Keyboard::Key::A))
        {
            if (dx != 1) {
                dx = -1;
                dy = 0;
            }
        }

        // Движение вправо по клавише D.
        if (Keyboard::isKeyPressed(Keyboard::Key::D))
        {
            if (dx != -1) {
                dx = 1;
                dy = 0;
            }
        }

        // По Escape завершаем игру.
        if (Keyboard::isKeyPressed(Keyboard::Key::Escape))
        {
            game_running = false;
        }

        // =========================
        // 3. ТАЙМЕР ДВИЖЕНИЯ
        // =========================

        // Узнаем, сколько секунд прошло с прошлого кадра,
        // и сразу перезапускаем таймер.
        float time = clock.restart().asSeconds();

        // Накопили это время в timer.
        timer += time;

        // Если прошло достаточно времени,
        // значит пора сделать один "шаг" змейки.
        if (timer > delay)
        {
            // Сбрасываем накопленное время,
            // чтобы начать отсчёт до следующего шага.
            timer = 0.f;

            // =========================
            // 4. СЧИТАЕМ НОВУЮ ПОЗИЦИЮ ГОЛОВЫ
            // =========================

            // Новая голова = старая голова + направление движения.
            int new_x = array_X[0] + dx;
            int new_y = array_Y[0] + dy;

            // =========================
            // 5. ПРОВЕРКА СТОЛКНОВЕНИЯ СО СТЕНОЙ
            // =========================

            // Если новая позиция вышла на рамку,
            // значит игра заканчивается.
            if (new_x <= 0 || new_x >= width - 1 ||
                new_y <= 0 || new_y >= height - 1)
            {
                game_running = false;
            }

            // =========================
            // 6. ПРОВЕРКА СТОЛКНОВЕНИЯ С СОБОЙ
            // =========================

            bool hit_self = false;

            // Проверяем все звенья тела, кроме головы.
            for (int i = 1; i < length; i++)
            {
                if (new_x == array_X[i] && new_y == array_Y[i])
                {
                    hit_self = true;
                    break;
                }
            }

            if (hit_self)
            {
                game_running = false;
            }

            // Дальнейшие действия делаем только если игра ещё не закончилась.
            if (game_running)
            {
                // =========================
                // 7. ПРОВЕРЯЕМ, СЪЕЛИ ЛИ ЯБЛОКО
                // =========================

                bool ate_apple = (new_x == X_apple && new_y == Y_apple);

                if (ate_apple)
                {
                    // Увеличиваем длину змейки на 1.
                    length++;

                    // Защита от выхода за пределы массивов.
                    if (length >= max_length)
                    {
                        game_running = false;
                    }
                    else
                    {
                        // Ставим новое яблоко.
                        // Важно: яблоко не должно появиться на змейке.
                        bool position_ok;
                        do
                        {
                            X_apple = rand() % (width - 2) + 1;
                            Y_apple = rand() % (height - 2) + 1;

                            position_ok = true;

                            // Проверяем, не попало ли яблоко на тело змейки.
                            for (int i = 0; i < length; i++)
                            {
                                if (X_apple == array_X[i] && Y_apple == array_Y[i])
                                {
                                    position_ok = false;
                                    break;
                                }
                            }
                        } while (!position_ok);
                    }
                }

                // =========================
                // 8. СДВИГАЕМ ВСЮ ЗМЕЙКУ
                // =========================

                // Идем с конца к началу,
                // чтобы не затереть координаты раньше времени.
                //
                // Например:
                // звено 4 берет координаты звена 3,
                // звено 3 берет координаты звена 2 и т.д.
                for (int i = length - 1; i >= 1; i--)
                {
                    array_X[i] = array_X[i - 1];
                    array_Y[i] = array_Y[i - 1];
                }

                // =========================
                // 9. ОБНОВЛЯЕМ ГОЛОВУ
                // =========================

                // После сдвига тела ставим голову в новую позицию.
                array_X[0] = new_x;
                array_Y[0] = new_y;
            }
        }

        // =========================
        // 10. РИСОВАНИЕ КАДРА
        // =========================

        // Очищаем окно черным цветом перед новой отрисовкой.
        window.clear(Color::Black);

        // ---------- РАМКА ----------
        // Будем рисовать стенки красными блоками.
        block.setFillColor(Color::Red);

        // Проходим по всем клеткам поля.
        for (int y = 0; y < height; y++)
        {
            for (int x = 0; x < width; x++)
            {
                // Если клетка находится по краю поля —
                // рисуем там элемент рамки.
                if (x == 0 || x == width - 1 || y == 0 || y == height - 1)
                {
                    // Перевод координат клетки в пиксели:
                    // экранный X = x * cell
                    // экранный Y = y * cell
                    block.setPosition({ x * cell, y * cell });

                    // Рисуем блок в окне.
                    window.draw(block);
                }
            }
        }

        // ---------- ЯБЛОКО ----------
        // Яблоко тоже рисуем блоком.
        block.setFillColor(Color::Red);
        block.setPosition({ X_apple * cell, Y_apple * cell });
        window.draw(block);

        // ---------- ТЕЛО ЗМЕЙКИ ----------
        // Тело рисуем зеленым цветом.
        // Начинаем с i = 1, потому что i = 0 — это голова.
        block.setFillColor(Color::Green);
        for (int i = 1; i < length; i++)
        {
            block.setPosition({ array_X[i] * cell, array_Y[i] * cell });
            window.draw(block);
        }

        // ---------- ГОЛОВА ЗМЕЙКИ ----------
        // Голову рисуем отдельно, другим цветом,
        // чтобы она отличалась от тела.
        block.setFillColor(Color::Yellow);
        block.setPosition({ array_X[0] * cell, array_Y[0] * cell });
        window.draw(block);

        // Показываем готовый кадр на экране.
        window.display();
    }

    return 0;
}
	```

# Улучшаем старый код
## Библиотеки и вывод
**Консольная версия:**
```c++
#include<iostream>
#include<windows.h>
#include<conio.h>
using namespace std;
```
<br>Раньше в консольной версии мы подключали `iostream`, `windows.h` и `conio.h`, потому что вся игра работала прямо в консоли. Через `iostream` мы выводили символы (`cout`), через `windows.h` управляли положением курсора, цветом текста и размером окна, а `conio.h` позволял обрабатывать нажатия клавиш без ожидания Enter (`_kbhit()`, `_getch()`). То есть вся логика отображения строилась вокруг текста: мы буквально «рисовали» игру символами, двигая курсор по экрану
**SFML версия:**
```c++
#include <SFML/Graphics.hpp>
#include <ctime>
#include <optional>

using namespace sf;
using namespace std;
```
Теперь мы подключаем `SFML/Graphics.hpp`, потому что переходим от текстового вывода к полноценной графике. SFML позволяет создавать окно, рисовать фигуры, задавать им цвет, размер и позицию в пикселях, а также удобно обрабатывать ввод с клавиатуры. Вместо того чтобы печатать символы, мы будем рисовать прямоугольники (`RectangleShape`), и игра будет выглядеть как настоящая 2D-игра. Дополнительно будем использовать `<ctime>` для случайных чисел и `<optional>` для удобной работы с событиями окна. Такой подход более современный и универсальный: код становится чище, игра работает плавнее, и её можно дальше развивать - добавлять анимации, текст, звуки и интерфейс.
## Размеры и масштаб
### Удаляем работу с консолью
Было:
```c++
    system("title Snake Game");
    system("mode con cols=70 lines=31");
    MoveWindow(GetConsoleWindow(), 50, 50, 800, 600, true);
```
**Что это делало:**
- `title` - менял заголовок окна консоли
- `mode` - задавал размер консоли (в символах)
- `MoveWindow` - двигал и менял размер окна
Все это мы полностью удаляем.  Потому что мы больше не работаем с консолью вообще. Теперь у нас своё графическое окно, и управляет им уже SFML
### Добавляем масштаб (cell) и окно SFML
```c++
    // Это уже есть в коде
    srand(time(0)); // чтобы яблоко появлялось случайно

    // размеры поля в клетках
    const int width = 50, height = 30;
    const int max_length = 50;
        
    // Это добавляем    
    const float cell = 20.f; // размер одной клетки в пикселях
    // Создаем графическое окно
    RenderWindow window(VideoMode({ (unsigned int)(width * cell), (unsigned int)(height * cell) }), "Snake Game");

```
`const float cell = 20.f;` - Переводит **логические координаты** в **пиксели **и позволяет управлять размером игры. 
Если `cell = 20`, значит каждая клетка змейки - это квадрат 20×20 пикселей. Если поставить 10 - всё станет меньше, если 40 - больше.
`RenderWindow window…` - Создаём **графическое окно. **Размер окна считается как: ширина = width \* cell, высота = height \* cell
Раньше у нас 1 клетка = 1 символ. Теперь 1 клетка = квадрат из пикселей (например 20×20)
- `RenderWindow` — это **тип объекта** из SFML, который представляет графическое окно.
- `window` — это **имя переменной**, которую мы будем использовать, чтобы управлять окном (рисовать, закрывать, проверять события).
- `VideoMode` — это **структура**, которая задаёт **размер окна** в пикселях.
- В скобках указываем ширину и высоту: `{ width * cell, height * cell }`
- `(unsigned int)` -  SFML **требует целое положительное число** типа `unsigned int` для ширины и высоты окна. У нас `width * cell` — это `float * float = float`, а не `unsigned int`. `(unsigned int)` — это **приведение типа**, чтобы компилятор не ругался. Коротко: Мы говорим SFML: “возьми эти числа, они должны быть целыми и положительными”
> Раньше мы управляли консолью — задавали её размер, двигали окно и рисовали символами. Теперь мы полностью отказались от консоли и создаём своё графическое окно через SFML. Добавили переменную `cell`, которая отвечает за размер клетки в пикселях. При этом сама логика игры (размер поля, координаты змейки) осталась прежней — мы просто начали по-другому это отображать.
## Игровая логика змейки (без изменений)
```c++
		// Массивы координат змейки
    int array_X[max_length]; // координата X каждого звена змейки
    int array_Y[max_length]; // координата Y каждого звена змейки
    int length = 1; // длина змейки

		// Начальная позиция змейки
    array_X[0] = width / 2; 
    array_Y[0] = height / 2;

		// Направление движения
    int dx = 1, dy = 0;

		// Позиция яблока
    int X_apple, Y_apple;
    do {
        X_apple = rand() % (width - 2) + 1;
        Y_apple = rand() % (height - 2) + 1;
    } while (X_apple == array_X[0] && Y_apple == array_Y[0]);
```
Этот кусок кода мы не меняем, потому что он отвечает не за графику, а за логику игры — где находится змейка, куда она движется и где появляется яблоко.
### Массивы координат змейки
```c++
int array_X[max_length];
int array_Y[max_length];
int length = 1;
```
- `array_X` и `array_Y` хранят **координаты каждой клетки змейки**.
- `length` — текущая длина змейки (изначально 1, только голова).
> Напоминаем ученикам: «Массивы — это как список точек на поле, где находится каждый сегмент змейки».
### Начальная позиция змейки
```c++
array_X[0] = width / 2;
array_Y[0] = height / 2;
```
- Ставим **голову змейки в центр поля**.
- `0` — индекс головы. Остальные индексы будут заполняться при увеличении длины змейки.
### Направление движения
```c++
int dx = 1, dy = 0;
```
- `dx`, `dy` — куда движется голова:
	- `dx = 1` → вправо,
	- `dy = 0` → нет вертикального движения.
- Позволяет легко менять направление при нажатии клавиш.
### Позиция яблока
```c++
int X_apple, Y_apple;
do {
    X_apple = rand() % (width - 2) + 1;
    Y_apple = rand() % (height - 2) + 1;
} while (X_apple == array_X[0] && Y_apple == array_Y[0]);
```
- `rand()` выбирает случайную позицию яблока **в пределах поля, не на границе**.
- `do ... while` — проверяем, чтобы яблоко **не появилось прямо на змейке**.
> Напоминаем: «Это случайная еда, которую змейка должна съесть, чтобы расти».
## Символы больше не нужны
Консольная версия:
```c++
    char snake = '*';
    char head = 2;
    char apple = 'o';
```
В консоли мы должны были выбирать символы, которыми рисовали объекты:
- `'*'` — тело змейки
- `2` — голова
- `'o'` — яблоко
То есть эти три строки мы просто удаляем.
**SFML версия:**
Таких переменных больше нет.
Почему? Потому что теперь мы не печатаем символы. Мы рисуем фигуры.
Вместо символов у нас будут:
- цвет
- форма
- позиция
- размер
Например:
- яблоко — красный квадрат
- тело змейки — зелёный квадрат
- голова — жёлтый квадрат
> В графических играх внешний вид объектов задаётся уже не символом, а тем, как мы их рисуем
### Курсор, цвет текста и `HANDLE` больше не нужны
```c++
    COORD c;
    HANDLE h = GetStdHandle(STD_OUTPUT_HANDLE);

    CONSOLE_CURSOR_INFO cci = { sizeof(cci), false };
    SetConsoleCursorInfo(h, &cci);
```
Раньше это нужно было для управления консолью:
- `COORD` — координаты курсора
- `HANDLE h` — доступ к консоли
- `SetConsoleCursorInfo` — скрывал мигающий курсор
Эти строки тоже просто удаляем
**SFML версия:**
Этого больше нет.
Почему? Потому что в графическом окне нет текстового курсора, который нужно двигать.
Мы просто рисуем фигуры в нужном месте.
