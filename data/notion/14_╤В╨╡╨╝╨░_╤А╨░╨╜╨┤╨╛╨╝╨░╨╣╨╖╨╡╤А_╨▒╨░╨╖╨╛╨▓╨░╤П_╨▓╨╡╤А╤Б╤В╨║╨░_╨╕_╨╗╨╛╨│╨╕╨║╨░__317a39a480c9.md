# Логика приложения
<callout icon="✅" color="gray_bg">
	Прописать логику работы приложения.
</callout>
- объявляем переменные
- создаем обработчик событий 
```kotlin
package com.yxl.calculator

import android.os.Bundle
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import android.widget.TextView
import android.widget.EditText
import android.widget.Button
import kotlin.random.Random

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        setContentView(R.layout.activity_main)
        enableEdgeToEdge()
        
				// Объявляем переменные	
        val etStart = findViewById<EditText>(R.id.etStart)
        val etEnd = findViewById<EditText>(R.id.etEnd)
        val bRandom = findViewById<Button>(R.id.bRandom)
        val tvResult = findViewById<TextView>(R.id.tvResult)

        // Ставим обработчик событий на кнопку
        bRandom.setOnClickListener {
            // Проверяем, заполнены ли все поля, если нет, то меняем надпись с результатом на ошибку
            if (etStart.text.toString().isEmpty() || etEnd.text.toString().isEmpty()) {
                tvResult.text = "Введите оба значения!"
            } else if (etStart.text.toString().toInt() > etEnd.text.toString().toInt()) {
                tvResult.text = "Начало должно быть меньше или равно концу!"
            } else {
                // Генерация случайного числа в диапазоне [начальное значение; конечное]
                val randomNumber = Random.nextInt(etStart.text.toString().toInt(), etEnd.text.toString().toInt() + 1)
                // Записываем рандомное число в нашу надпись с результатом
                tvResult.text = randomNumber.toString()
			            }
            }
        }
    }
}
```
